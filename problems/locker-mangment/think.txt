interface lockerAllocationsStrategy{
    allocate(package:package)
}


class randomAllocations implements lockerAllocationsStrategy{
    allocate(package:package,lockers:locker){
        lockers[math.floor(math.random() * lockers.length)]
    }
}



class lockerServices(){
    private lockers:locker = locker[]
    private lockerAllocationsStrategy:lockerAllocationsStrategy

    // constructor
    constructor(lockerAllocationsStrategy:lockerAllocationsStrategy){
        this.lockerAllocationsStrategy = lockerAllocationsStrategy
    }
    
    // create locker
    void createLocker(locker_id,locker_size,is_vacate,occupied_time){
        let locker:locker = new locker(locker_id,locker_size,is_vacate,occupied_time)
        this.lockers.push(locker)
    }

    void vacateLocker(locker_id:int){
        let locker  =  this.lockers.filter(locker=>locker.locker_id = locker_id)
        locker.is_vacate = true,
        locker.occupied_time = null
        
    }

    // get all the vacate locker 
    locker getAllVacateLocker(){
        return this.lockers.filter(locker=>locker.is_vacate)
    }

    // allocate locker
    locker allocateLocker(package){
        let vacate_locker = this.getAllVacateLocker()
        let locker =  this.lockerAllocationsStrategy(package,vacate_locker)
        locker.is_vacate = false,
        locker.occupied_time = Date.now(),
        return locker
    }

    // overdueLocker 
    locker overdueLocker(){
        return this.lockers.filter(locker=>locker.occupied_time + 3 days > date.now())
    }
}
